% Command for colored cell with number
\newcommand{\scoreCell}[1]{%
  \ifnum#1=10
    \cellcolor{score10}#1
  \else\ifnum#1=8
    \cellcolor{score8}#1
  \else\ifnum#1=6
    \cellcolor{score6}#1
  \else\ifnum#1=4
    \cellcolor{score4}#1
  \else\ifnum#1=2
    \cellcolor{score2}#1
  \fi\fi\fi\fi\fi
}
